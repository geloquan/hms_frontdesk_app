select 
	op.label as operation_label,
    opt.on_site as is_tool_on_site,
    t.status as tool_status,
    e.name as tool_name
from 
operationtool opt
left join
operation op on opt.operation_id = op.id
left join 
tool t on opt.tool_id = t.id
left join
patient p on op.patient_id = p.id
left join
room r on op.room_id = r.id
left join
equipment e on t.info_id = e.id

where op.id = 1
